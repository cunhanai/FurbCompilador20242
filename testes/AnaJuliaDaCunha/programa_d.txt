main
    i_mes, s_trimestre;

    read(i_mes);

    repeat
        if i_mes < 1 || i_mes > 12
            read("valor invalido, digitar um valor entre 1 e 12: ", i_mes);
        end;
    while i_mes < 1 || i_mes > 12;

    if i_mes < 4
        s_trimestre = "primeiro trimestre";
    elif i_mes < 7
        s_trimestre = "segundo trimestre";
    elif i_mes < 10
        s_trimestre = "terceiro trimestre";
    else
        s_trimestre = "quarto trimestre";
    end;

    writeln(s_trimestre);
end
